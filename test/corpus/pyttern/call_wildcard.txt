================================================================================
Simple call wildcard
================================================================================

?$MySubpattern()

--------------------------------------------------------------------------------

(module
  (expression_statement
    (call_wildcard
      name: (identifier))))

================================================================================
Call wildcard with arguments
================================================================================

?$MySubpattern(x, y=1)

--------------------------------------------------------------------------------

(module
  (expression_statement
    (call_wildcard
      name: (identifier)
      (subpattern_args
        (subpattern_arg
          (identifier))
        (subpattern_arg
          (identifier)
          (integer))))))

================================================================================
Call wildcard as expression
================================================================================

a = ?$MySubpattern(?val)

--------------------------------------------------------------------------------

(module
  (expression_statement
    (assignment
      left: (identifier)
      right: (call_wildcard
        name: (identifier)
        (subpattern_args
          (subpattern_arg
            (identifier
              (var_wildcard
                name: (identifier)))))))))
